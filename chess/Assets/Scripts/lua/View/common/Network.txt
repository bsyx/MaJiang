Network = {};
local this = Network;

local transform;
local gameObject;

function Network.Start() 
    log("Network.Start!!");
    NetManager.OnInit();
end

--Socket消息--
function Network.OnSocket(key, data)
    --关闭loading加载界面
    log("=======key====="..tostring(key));
    EventManager.InvokeEvent(SetLoadActive,false);
    EventManager.InvokeEvent(tostring(key), data);
end

--卸载网络监听--
function Network.Unload()
    NetManager.ClearCallBacks();
end
